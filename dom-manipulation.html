Class Manipulation

Usage Syntax

// Add class
ok('.element').class('active');

// Remove class
ok('.element').unclass('active');

// Toggle class
ok('.element').toggleClass('active');

CSS Manipulation

Usage Syntax

// Set single CSS property
ok('.element').css('color', 'red');

// Set multiple CSS properties
ok('.element').css({
  'color': 'red',
  'font-size': '16px'
});

// Get CSS property
const color = ok('.element').css('color');
